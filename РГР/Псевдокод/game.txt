
class Карточка :
Описание(_Описание)
{
    public:
        Карточка(string _Описание);
        operator<<(Карточка* this){
            cout << Описание << endl;
        }
    private:
        string Описание
}

class Эмоция : public Карточка :
Описание(_Описание),
Название(_Название)
{
    Эмоция(string _Название, string _Описание);
    operator<<(Карточка* this){
        cout << Название << endl;
        cout << Описание << endl;
    }
    private:
        string Название
}

class Профессия : public Карточка
{
    public:
        Профессия(string _Описание, vector<int> _НомПрет);
        operator<<(Карточка* this){
            cout << Описание << endl;
        }
        vector<int> ПолучитьПретендентов(){
            return НомераПретендентов
        }
    private:
        vector<int> НомераПретендентов
}
